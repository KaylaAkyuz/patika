## Install

```bash
npm install --save sakura-button
```

## Usage

```jsx
import React, { Component } from 'react'

import { Button } from 'sakura-button'
import 'sakura-button/dist/index.css'

class Example extends Component {
  render() {
    return <Button />
  }
}
```
